// ============================================================ // ALLMYTOUR 인사평가 Apps Script // ============================================================ const SHEET_SELF = '인사평가_셀프'; const SHEET_DOWN = '인사평가_하향'; const SHEET_PROBATION = '인사평가_수습'; const SHEET_TARGETS = '평가대상자'; const SHEET_ADMIN = '인사평가 접근권한'; // ── JSON 응답 (JSONP callback 지원) ────────────────────────── function jsonResponse(obj, callback) { const json = JSON.stringify(obj); if (callback) { return ContentService .createTextOutput(callback + '(' + json + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(json) .setMimeType(ContentService.MimeType.JSON); } // ── base64 디코딩 ───────────────────────────────────────────── function b64decode(str) { if (!str) return ''; try { return decodeURIComponent(escape(Utilities.base64Decode(str, Utilities.Charset.UTF_8).map(b => String.fromCharCode(b)).join(''))); } catch(e) { try { return Utilities.newBlob(Utilities.base64Decode(str)).getDataAsString('UTF-8'); } catch(e2) { return str; } } } // ── HTML 태그 제거 ──────────────────────────────────────────── function strip(html) { return (html || '').replace(/<[^>]*>/g, '').replace(/ /g, ' ').replace(/\s+/g, ' ').trim(); } // ── GET 처리 ────────────────────────────────────────────────── function doGet(e) { const p = e.parameter || {}; const action = p.action || ''; const callback = p.callback || ''; // JSONP 콜백 함수명 try { if (action === 'save_self') return handleSaveSelf(p); if (action === 'save_down') return handleSaveDown(p); if (action === 'save_probation') return handleSaveProbation(p); if (action === 'query') return handleQuery(p); if (action === 'get_cases') return jsonResponse({ success: true, cases: {} }, callback); if (action === 'admin_all') return handleAdminAll(callback); if (action === 'check_admin') return handleCheckAdmin(p, callback); if (action === 'get_salary_list') return handleGetSalaryList(); return jsonResponse({ success: false, error: 'Unknown: ' + action }, callback); } catch(err) { return jsonResponse({ success: false, error: err.message }, callback); } } // ── POST 처리 (긴 사례 데이터 저장용) ─────────────────────── function doPost(e) { try { const p = JSON.parse(e.postData.contents || '{}'); const action = p.action || ''; if (action === 'save_self') return handleSaveSelf(p); if (action === 'save_down') return handleSaveDown(p); if (action === 'save_probation') return handleSaveProbation(p); return jsonResponse({ success: false, error: 'Unknown action: ' + action }); } catch(err) { return jsonResponse({ success: false, error: err.message }); } } // ── 시트 헬퍼 ──────────────────────────────────────────────── function getOrCreateSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); return ss.getSheetByName(name) || ss.insertSheet(name); } function initHeaders(sheet, headers, color) { if (sheet.getLastRow() === 0) { sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length) .setFontWeight('bold').setBackground(color || '#F1F5F9'); } } function nowStr() { return Utilities.formatDate(new Date(), 'Asia/Seoul', 'yyyy. M. d. a h:mm:ss'); } // ── 셀프리뷰 저장 ───────────────────────────────────────────── function handleSaveSelf(p) { const sheet = getOrCreateSheet(SHEET_SELF); initHeaders(sheet, ['제출일시','성명','사번','부서','직위','담당업무','입사일자','작성일자','셀프점수','서술형','result_url']); let casesText = ''; // POST(JSON)로 오면 cases가 객체, GET(b64)으로 오면 문자열 if (p.cases && typeof p.cases === 'object') { casesText = Object.entries(p.cases).map(([k, v]) => (v ? v : '')).filter(Boolean).join('\n\n'); } else if (p.cases) { const casesRaw = b64decode(p.cases); try { const caseObj = JSON.parse(casesRaw); casesText = Object.entries(caseObj).map(([k, v]) => (v ? v : '')).filter(Boolean).join('\n\n'); } catch(e) { casesText = casesRaw; } } sheet.appendRow([nowStr(), p.name||'', p.employee_id||'', p.dept||'', p.pos||'', p.work||'', p.join||'', p.date||'', p.self_score||'0', casesText, p.result_url||'']); return jsonResponse({ success: true }); } // ── 하향리뷰 저장 ───────────────────────────────────────────── function handleSaveDown(p) { const sheet = getOrCreateSheet(SHEET_DOWN); initHeaders(sheet, ['제출일시','평가자','평가대상자','사번','부서','직위','담당업무','입사일자','작성일자','하향점수','서술형','result_url']); let casesText = ''; if (p.cases && typeof p.cases === 'object') { casesText = Object.entries(p.cases).map(([k, v]) => (v ? v : '')).filter(Boolean).join('\n\n'); } else if (p.cases) { const casesRaw = b64decode(p.cases); try { const caseObj = JSON.parse(casesRaw); casesText = Object.entries(caseObj).map(([k, v]) => (v ? v : '')).filter(Boolean).join('\n\n'); } catch(e) { casesText = casesRaw; } } sheet.appendRow([nowStr(), p.evaluator||'', p.name||'', p.employee_id||'', p.dept||'', p.pos||'', p.work||'', p.join||'', p.date||'', p.down_score||p.self_score||'0', casesText, p.result_url||'']); return jsonResponse({ success: true }); } // ── 수습리뷰 저장 ──────────────────────────────────────────── function handleSaveProbation(p) { const sheet = getOrCreateSheet(SHEET_PROBATION); initHeaders(sheet, ['제출일시','평가자','평가대상자','사번','부서','직위','담당업무', '입사일자','작성일자','수습점수','역할및업무','활동및실적','강점','종합소견','result_url'], '#F0FDF4'); sheet.appendRow([ nowStr(), p.evaluator||'', p.name||'', p.employee_id||'', p.dept||'', p.pos||'', p.work||'', p.join||'', p.date||'', p.prob_score||'0', strip(b64decode(p.prob_q1)), strip(b64decode(p.prob_q2)), strip(b64decode(p.prob_q3)), strip(b64decode(p.prob_q4)), p.result_url||'' ]); return jsonResponse({ success: true }); } // ── 셀프리뷰 조회 ───────────────────────────────────────────── function handleQuery(p) { const name = p.name || ''; if (!name) return jsonResponse({ success: false, error: '이름 없음' }); const sheet = getOrCreateSheet(SHEET_SELF); const data = sheet.getDataRange().getValues(); if (data.length < 2) return jsonResponse({ success: false, error: '데이터 없음' }); const h = data[0]; const ni = h.indexOf('성명'), si = h.indexOf('셀프점수'), ui = h.indexOf('result_url'); let found = null; for (let i = data.length - 1; i >= 1; i--) { if (String(data[i][ni]) === name) { found = data[i]; break; } } if (!found) return jsonResponse({ success: false, error: '데이터 없음' }); return jsonResponse({ success: true, name: found[ni], self_score: found[si], result_url: found[ui], cases: {} }); } // ── 관리자 전체 조회 ────────────────────────────────────────── function handleAdminAll(callback) { return jsonResponse({ success: true, self: getRows(SHEET_SELF, ['제출일시','성명','부서','직위','셀프점수','result_url']), down: getRows(SHEET_DOWN, ['제출일시','평가자','평가대상자','부서','하향점수','result_url']), probation: getRows(SHEET_PROBATION, ['제출일시','평가자','평가대상자','부서','수습점수','result_url']) }, callback); } function getRows(sheetName, cols) { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName); if (!sheet) return []; const data = sheet.getDataRange().getValues(); if (data.length < 2) return []; const h = data[0]; return data.slice(1) .map(row => { const obj = {}; cols.forEach(c => { obj[c] = String(row[h.indexOf(c)] || ''); }); return obj; }) .filter(r => r['성명'] || r['평가대상자']); } // ── 관리자 인증 ─────────────────────────────────────────────── function handleCheckAdmin(p, callback) { const inputId = (p.name || '').trim(); const inputPw = (p.pw || '').trim(); if (!inputId || !inputPw) return jsonResponse({ success: false, error: '이름과 비밀번호를 입력하세요' }, callback); const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_ADMIN); if (!sheet) return jsonResponse({ success: false, error: '관리자 시트를 찾을 수 없습니다' }, callback); const data = sheet.getDataRange().getValues(); if (data.length < 2) return jsonResponse({ success: false, error: '관리자 데이터 없음' }, callback); // A=이름, B=비밀번호 (숫자/문자 모두 문자열로 비교, 1행=헤더 건너뜀) for (let i = 1; i < data.length; i++) { const rowId = String(data[i][0] ?? '').trim(); const rowPw = String(data[i][1] ?? '').trim(); if (!rowId) continue; // 빈 행 건너뜀 if (rowId === inputId && rowPw === inputPw) { return jsonResponse({ success: true, name: rowId }, callback); } } return jsonResponse({ success: false, error: '비밀번호가 올바르지 않습니다' }, callback); } // ── 평가 대상자 목록 ────────────────────────────────────────── function handleGetSalaryList() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_TARGETS); if (!sheet) return jsonResponse({ success: true, targets: [] }); const data = sheet.getDataRange().getValues(); if (data.length < 2) return jsonResponse({ success: true, targets: [] }); const h = data[0]; const ni = h.indexOf('성명'), di = h.indexOf('부서'), ji = h.indexOf('입사일자'), si = h.indexOf('재직상태'); return jsonResponse({ success: true, targets: data.slice(1) .filter(r => r[ni]) .map(r => ({ name: String(r[ni]||''), dept: String(r[di]||''), join: String(r[ji]||''), status: String(r[si]||'재직') })) }); }
